include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(${CMAKE_BINARY_DIR})
include_directories(${CMAKE_SOURCE_DIR}/mindspore/ops)
include_directories(${CMAKE_SOURCE_DIR}/mindspore/ccsrc)
include_directories(${CMAKE_SOURCE_DIR}/mindspore/core/include)

file(GLOB RES_MANAGER_LIST RELATIVE ${CMAKE_CURRENT_SOURCE_DIR}
        "*.cc")
add_subdirectory(${CMAKE_SOURCE_DIR}/mindspore/ccsrc/runtime/device/res_manager/utils)
list(APPEND RES_MANAGER_LIST $<TARGET_OBJECTS:_mindspore_runtime_device_res_manager_utils_obj>)
if(CMAKE_SYSTEM_NAME MATCHES "Windows")
        target_compile_definitions(_mindspore_runtime_device_res_manager_utils_obj PRIVATE RES_MANAGER_DLL)
endif()
set_property(SOURCE ${RES_MANAGER_LIST} PROPERTY COMPILE_DEFINITIONS SUBMODULE_ID=mindspore::SubModuleId::SM_DEVICE)
add_library(mindspore_res_manager SHARED ${RES_MANAGER_LIST})
target_link_libraries(mindspore_res_manager PRIVATE mindspore_core mindspore_ops mindspore_common
        mindspore_ops_kernel_common PRIVATE mindspore_memory_pool)
if(CMAKE_SYSTEM_NAME MATCHES "Windows")
        target_compile_definitions(mindspore_res_manager PRIVATE RES_MANAGER_DLL)
endif()
set_target_properties(mindspore_res_manager PROPERTIES INSTALL_RPATH $ORIGIN)
